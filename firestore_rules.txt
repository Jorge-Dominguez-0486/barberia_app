rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Función para verificar si es admin
    function isAdmin() {
      return request.auth != null && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }
    
    // Usuarios: cada quien puede leer/editar su propio perfil; admin puede todo
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth.uid == userId || isAdmin();
    }
    
    // Productos: todos los usuarios autenticados pueden leer; solo admin puede escribir
    match /products/{productId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }
    
    // Servicios: todos los usuarios autenticados pueden leer; solo admin puede escribir
    match /services/{serviceId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }
    
    // Categorías: todos los usuarios autenticados pueden leer; solo admin puede escribir
    match /categories/{categoryId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }
    
    // Citas: el cliente puede leer/crear las suyas; admin puede todo
    match /appointments/{appointmentId} {
      allow read: if request.auth.uid == resource.data.clienteId || isAdmin();
      allow create: if request.auth != null;
      allow update, delete: if request.auth.uid == resource.data.clienteId || isAdmin();
    }
  }
}
